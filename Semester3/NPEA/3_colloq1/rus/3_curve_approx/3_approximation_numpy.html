import numpy as np 
import math

def solve(x, y):
    if (len(x) != len(y)):
        raise RuntimeError('len(x) != len(y)')
    
    p = []
    degree = 0
    err = math.inf

    while err >= 0.01 :
        degree += 1
        p = np.polyfit(x, y, degree)
        y_approx = np.polyval(p, x)
        err = np.max(abs( (y_approx - y) / y * 100 ))

    return p, degree

x = np.array([2, 3, 4, 5, 7, 9, 10, 11, 13, 14])
y = np.array([0.25, 1.25, 0.1, 3.25, 3, 2, 4, 5.5, 6.5, 5.75])
